<template>
    <div>
        <div class="header-info">
            <p>{{action_name}}-${title}</p>
            <div>
                <el-button size="mini" @click="$router.push('/${group}/${entity_url}/add')" v-if="action=='edit'">新建</el-button>
                <el-button size="mini" @click="save_click" type="primary"> 保存</el-button>
            </div>
        </div>
        <el-row :gutter="12" type="flex">
            <el-col>
                <el-card shadow="always">
                    <div slot="header">
                        <span>基本信息</span>
                    </div>
${for:fields}
${fif:#enum}
                    <kv label="${remark}">
                       <selector type="radio" v-model="info.${name}" enum="${type}" />
                    </kv>
${elseif:@IdUrl}
                    <kv>
                        <label slot="k">${remark}</label>
                        <upload
                            :maxCount="1"
                            v-model="info.${name}"
                            fileType="img"
                            scales="16:9"
                            :maxWidth="1024"
                            maxSize="5M"
                        ></upload>
                    </kv>
${elseif:@IdName}
                    <kv label="${remark}">
                       <ref-${name} v-model="info.${name}"></ref-${name}>
                    </kv>
${elseif:@Boolean}
                    <kv label="${remark}">
                       <selector  type="radio" v-model="info.${name}" :data="{true:'是',false:'否','':'全部'}" />
                    </kv>
${elseif:@LocalDate}
                    <el-date-picker v-model="info.${name}" placeholder="选择日期" />
${elseif:@LocalDateTime}
                    <el-date-picker v-model="info.${name}" placeholder="选择日期时间"  type="datetime" />
${elseif:@LocalTime}
                    <el-time-select v-model="info.${name}" placeholder="选择时间" />
${else}
                    <kv label="${remark}"><el-input v-model="info.${name}"/></kv>
${endif}
${endfor}
                </el-card>
            </el-col>
            <el-col>
                <el-card shadow="always">
                    <div slot="header">
                        <span>扩展信息</span>
                    </div>
                </el-card>
            </el-col>
        </el-row>
    </div>
</template>
<style scoped>
</style>
<script>
/**
 * Created by CodeGenerator
 */
${for:fields}
${fif:@IdName}
        import ref${name} from "../home/EmptyRef"
${endif}
${endfor}
    export default {
        components: {
${for:fields}
${fif:@IdName}
        "ref-${name}": ref${name},
${endif}
${endfor}
        },
        data() {
            return {
${for:fields}
${fif:#enum}
                ${type}: jv.${type}.getData(),
${endif}
${endfor}
                //URL参数
                action: "",
                action_name: "",
                id: "",
                info: {}, //子对象需要声明。
            }
        },
        watch:{
           action(val){
               this.action_name =  {add: "添加", edit: "修改"}[val]
           }
        },
        mounted() {
            this.action = this.$route.params.action;
            this.id = this.$route.query.id;

            this.loadData()
        },
        methods: {
            loadData() {
                if (!this.id) return;
                this.$http.post("/${group}/${entity_url}/detail?id=" + this.id).then(res => {
                    this.info = res.data.data;
                });
            },
            save_click() {
                //校验
                if (jv.main.chk() == false) {
                    return;
                }

                this.$http.post("/${group}/${entity_url}/save", this.info).then(res => {
                    jv.info(this.action_name + " 成功");
                    if (this.action == "add") {
                        this.$router.push("/${group}/${entity_url}/edit?id=" + res.data.data)
                    }
                })
            },

        }
    }
</script>
